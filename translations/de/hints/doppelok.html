def checkio(expression: str) -> bool:

    brackets = {'(': ')', '[': ']', '{': '}'}
    new = ['']
    for i in expression:
        if i in brackets:
            new.append(brackets[i])

Mit Hilfe des Generators lassen wir lassen wir nur Klammern in der Zeile stehen. Und dann, bei jedem Schritt, bereinigen wir die Zeile von Kombinationen von Klammern. Wenn die Zeile auf Null bereinigt wurde, geben wir True zurück, und wenn eine falsche Kombination übrig bleibt, geben wir False zurück

def checkio(expression: str) -> bool:

    s = ''.join(c for c in expression if c in '([{}])')
    while s:
        s0, s = s, s.replace('()', '').replace('[]', '').replace('{}', '')
        if s == s0:
            return False
    return True

def checkio(expr):

    for c in '1234567890+-*/':
        expr = expr.replace(c, '')
    length = len(expr)
    while length:
        if expr.count('()'):
            expr = expr.replace('()','')
        if expr.count('{}'):
            expr = expr.replace('{}','')
        if expr.count('[]'):
            expr = expr.replace('[]','')
        if len(expr) == length:
            break
        else:
            length = len(expr)
    return len(expr) == 0
