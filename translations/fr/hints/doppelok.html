def checkio(expression: str) -> bool:

    s = ''.join(c for c in expression if c in '([{}])')
    while s:
        s0, s = s, s.replace('()', '').replace('[]', '').replace('{}', '')
        if s == s0:
            return False
    return True

def checkio(expr):

    for c in '1234567890+-*/':
        expr = expr.replace(c, '')
    length = len(expr)
    while length:
        if expr.count('()'):
            expr = expr.replace('()','')
        if expr.count('{}'):
            expr = expr.replace('{}','')
        if expr.count('[]'):
            expr = expr.replace('[]','')
        if len(expr) == length:
            break
        else:
            length = len(expr)
    return len(expr) == 0
