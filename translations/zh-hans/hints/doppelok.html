好吧，好吧！这就是精神！=) 下面是通过一个while循环和一个发生器表达式实现的解决方案。使用生成器，我们 行中只留下括号。然后，每一步，我们都会清除行中的括号组合。 如果该行被清除为零，我们返回True，如果仍然存在错误的组合，我们返回False。

def checkio(expression: str) -> bool:

    s = ''.join(c for c in expression if c in '([{}])')
    while s:
        s0, s = s, s.replace('()', '').replace('[]', '').replace('{}', '')
        if s == s0:
            return False
    return True

def checkio(expr):

    for c in '1234567890+-*/':
        expr = expr.replace(c, '')
    length = len(expr)
    while length:
        if expr.count('()'):
            expr = expr.replace('()','')
        if expr.count('{}'):
            expr = expr.replace('{}','')
        if expr.count('[]'):
            expr = expr.replace('[]','')
        if len(expr) == length:
            break
        else:
            length = len(expr)
    return len(expr) == 0
